```typescript
/**
 * Dev-mode identity helpers. Fixed-seed identities for testing.
 * No circuit artifacts needed — only circomlibjs Poseidon + EdDSA.
 */
import { createHumanIdentity, createAgentCredential } from './identity';
import type { HumanIdentity, AgentCredential } from './types';

/** Fixed dev seed — deterministic, recognizable. */
const DEV_SECRET = BigInt('0xDEV0DEADBEEF');
const DEV_MODEL_HASH = BigInt('0xDEV0MODEL');

export interface DevIdentityOptions {
  permissionBitmask?: bigint;
  expiryTimestamp?: bigint;
}

export interface DevIdentities {
  human: HumanIdentity;
  agent: AgentCredential;
  operatorKey: Buffer;
}

let _warned = false;

export async function createDevIdentities(
  options?: DevIdentityOptions,
): Promise<DevIdentities> {
  if (!_warned) {
    console.warn(
      '⚠ Bolyra dev mode — identities are for testing only. Do not use in production.',
    );
    _warned = true;
  }

  const human = await createHumanIdentity(DEV_SECRET);

  // Deterministic operator key (32 bytes from fixed seed).
  const operatorKey = Buffer.alloc(32);
  operatorKey.writeUInt32BE(0xDEV0CAFE, 0);
  operatorKey.writeUInt32BE(0xDEV0BEEF, 4);

  const permissionBitmask = options?.permissionBitmask ?? 0b11111111n;
  const expiryTimestamp =
    options?.expiryTimestamp ??
    BigInt(Math.floor(Date.now() / 1000)) + 86400n;

  const agent = await createAgentCredential(
    DEV_MODEL_HASH,
    operatorKey,
    permissionBitmask,
    expiryTimestamp,
  );

  return { human, agent, operatorKey };
}
```

Note: `createAgentCredential` already accepts `(modelHash, operatorPrivateKey, permissionBitmask, expiryTimestamp)` per `sdk/src/identity.ts:102`. Verify the signature matches before implementing. If `createAgentCredential` takes a permissions array instead of a bitmask, use `permissionsToBitmask()` or pass the bitmask directly depending on the overload.

```typescript
async function verifyDevBundle(
  bundle: BolyraProofBundle,
  config: BolyraMcpConfig,
): Promise<BolyraAuthContext> {
  const network = config.network ?? DEFAULTS.network;
  const minScore = config.minScore ?? DEFAULTS.minScore;
  const maxProofAge = config.maxProofAge ?? DEFAULTS.maxProofAge;

  let nonce: bigint;
  let commitment: bigint;
  try {
    nonce = toBigInt(bundle.nonce, 'nonce');
    commitment = toBigInt(bundle.credentialCommitment, 'credentialCommitment');
  } catch (e: unknown) {
    return failCtx(e instanceof Error ? e.message : String(e));
  }

  // Synthetic credential from bundle signals
  const permissionBitmask = BigInt(bundle.agentProof.publicSignals[3] ?? '0');
  const expiryTimestamp = BigInt(bundle.agentProof.publicSignals[4] ?? '0');

  // Score (same rubric as production minus ZKP component)
  const warnings: string[] = [];
  let score = 40; // ZKP "passes" in dev mode

  const now = BigInt(Math.floor(Date.now() / 1000));
  if (expiryTimestamp > now) {
    score += 20;
  } else {
    warnings.push(`Credential expired at ${expiryTimestamp} (current: ${now})`);
  }

  if ((permissionBitmask & 0b11n) !== 0n) {
    score += 20;
  } else {
    warnings.push('Agent has no read/write permissions');
  }

  const nonceAgeSec = Number(now - nonce);
  if (nonceAgeSec >= 0 && nonceAgeSec < maxProofAge) {
    score += 10;
  } else {
    warnings.push(`Session nonce stale (${nonceAgeSec}s old, max ${maxProofAge}s)`);
  }

  score += 10; // Scope commitment "passes" in dev mode

  // Dev delegation chain: walk shape, skip crypto
  let chainDepth = 0;
  let effectivePermission = permissionBitmask;
  let effectiveCommitment = bundle.credentialCommitment;
  if (bundle.delegationChain && bundle.delegationChain.length > 0) {
    for (const link of bundle.delegationChain) {
      chainDepth++;
      effectivePermission = BigInt(link.delegateeScope);
      effectiveCommitment = link.delegateeCommitment;
    }
  }

  const passed = score >= minScore;
  return {
    verified: passed,
    score,
    did: buildDid(`dev`, commitment),
    permissionBitmask: effectivePermission,
    warnings,
    reason: passed ? undefined : warnings[0] ?? 'Dev verification failed',
    chainDepth,
    effectiveCommitment,
  };
}

function failCtx(reason: string): BolyraAuthContext {
  return {
    verified: false, score: 0, did: '', permissionBitmask: 0n,
    warnings: [], reason, chainDepth: 0, effectiveCommitment: '',
  };
}
```
